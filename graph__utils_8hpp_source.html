272 Path<Graph> build_path_from(std::vector<Vertex> const &path, Graph const &G,

276  typename property_map<Graph, edge_weight_t>::type>::value_type;
277  using Edge = typename graph_traits<Graph>::edge_descriptor;
278 
279  auto u = *path.begin();
280  auto len = Length{};
281 
282  auto path_es = std::unordered_set<Edge, boost::hash<Edge>>{};

284  auto v = *it;
285  auto [e, is_ok] = edge(u, v, G);
286  assert(is_ok && "[arlib::details::build_path_from] Edge not found.");
287  auto weight = W[e];

289  path_es.insert(e);

294  path.cend()};

298  alternative_path_vertices<Vertex>>;

349 template <typename Graph, typename WeightMap,
350  typename Vertex = vertex_of_t<Graph>>
351 std::vector<Path<Graph>> to_paths(Graph const &G,
352  multi_predecessor_map<Vertex> &pmap,
353  WeightMap const &weight, Vertex s, Vertex t) {
354  auto res = std::vector<Path<Graph>>{};
355  auto Q = std::stack<std::pair<int, std::vector<Vertex>>>{};
356  for (auto const &[k_th, v] : get(pmap, t)) {
357  Q.push({k_th, {t}});
358  }
359 
360  while (!Q.empty()) {
361  auto [k_th, path] = std::move(Q.top());
362  Q.pop();
363  if (path.front() != s) {
364  auto const &preds = get(pmap, path.front());
365  // Stack a pair only for predecessor of same k_th
366  for (auto const &[k_prime, pred] : preds) {
367  if (k_prime == k_th) {
368  auto path_prime = path;
369  path_prime.insert(path_prime.begin(), pred);
370  Q.push({k_prime, std::move(path_prime)});
371  }
372  }
373  } else {
374  res.push_back(details::build_path_from(path, G, weight));
375  }
376  }
377 
378  std::sort(res.begin(), res.end(), [](auto const &a, auto const &b) {
379  return a.length() < b.length();
380  });
381  return res;
382 }
