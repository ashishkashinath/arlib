136 template <typename Vertex, typename PredecessorMap, typename BackPredecessorMap>
137 void merge_paths(Vertex s, Vertex t, Vertex w, PredecessorMap predecessor_f,
138  BackPredecessorMap predecessor_b,
139  std::deque<Vertex> &final_path) {
140  final_path.clear();
141  final_path.push_front(w);
142  // Fill with forward partial result
143  auto cur_f = w;
144  while (cur_f != s) {
145  auto pred_f = predecessor_f[cur_f];
146  assert(pred_f != cur_f); // otherwise cur_f wouldnt be reachable, but if
147  // cur_f wouldnt be reachable we shouldnt be here.
148  final_path.push_front(pred_f);
149  cur_f = pred_f;
150  }
151 
152  auto cur_b = w;
153  while (cur_b != t) {
154  auto pred_b = predecessor_b[cur_b];
155  assert(pred_b != cur_b); // otherwise cur_b wouldnt be reachable, but if
156  // cur_b wouldnt be reachable we shouldnt be here.
157  final_path.push_back(pred_b);
158  cur_b = pred_b;
159  }

183 template <typename Graph, typename PredecessorMap, typename DistanceMap,
184  typename WeightMap, typename OtherPredecessorMap,
185  typename OtherDistanceMap, typename BiDijkstraVisitorImpl,
186  typename Vertex = vertex_of_t<Graph>,
187  typename Length = value_of_t<DistanceMap>>
188 BiDijkStepRes bi_dijkstra_step(
189  const Graph &G, Vertex s, Vertex t, PredecessorMap predecessor,
190  DistanceMap distance, WeightMap weight, Fringe<Vertex, Length> &fringe,
191  Seen<Vertex, Length> &seen, OtherPredecessorMap other_predecessor,
192  OtherDistanceMap other_distance, Fringe<Vertex, Length> &other_fringe,
193  Seen<Vertex, Length> &other_seen, Direction direction,
194  Length &final_distance, std::deque<Vertex> &final_path,
195  BiDijkstraVisitor<BiDijkstraVisitorImpl> &visitor) {
196  constexpr Length inf = std::numeric_limits<Length>::max();
197  // Extract closest node to expand
198  const auto [v, dist] = fringe.top();
199  fringe.pop();
200 
201  if (distance[v] != inf) {
202  // Shortest path to 'node' already found. Continue.
203  return BiDijkStepRes::next;
204  }
205 
206  // Ask the visitor if vertex should be expanded
207  auto lower_bound_v = other_fringe.top().second;
208  if (!visitor.expand_vertex(v, dist, lower_bound_v, final_distance)) {
209  return BiDijkStepRes::next;
210  }
211 
212  // Update distance
213  distance[v] = dist; // Equal to seen[v]
214  if (other_distance[v] != inf) {
215  // If we have scanned v in both directions we are done,
216  // we have now discovered the shortest path. But the visitor may ask to keep
217  // on searching.
218 
219  // Check terminating condition:
220  auto min_dist = fringe.top().second;
221  auto other_min_dist = other_fringe.top().second;
222 
223  if (visitor.terminating_condition(min_dist, other_min_dist,
224  final_distance)) {
225  return BiDijkStepRes::end;
226  }
227  // Please refer to:
228  // Andreas Paraskevopoulos, Christos Zaroliagis. Improved Alternative Route
229  // Planning. Daniele Frigioni and Sebastian Stiller. ATMOS - 13th Workshop
230  // on Algorithmic Approaches for Transportation Modelling, Optimizations and
231  // Systems - 2013
232  // auto terminating_condition = min_dist + other_min_dist > d_s_t;
233  }
234 
235  // Compute neighbor distances
236  for (auto [it, end] = out_edges(v, G); it != end; ++it) {
237  using boost::get;
238  auto w = target(*it, G);
239  auto min_weight = get(weight, *it);
240  auto vw_length = distance[v] + min_weight;
241 
242  if (distance[w] != inf) {
243  if (vw_length < distance[v]) {
244  // Throw some exception "Contradictory paths found: negative weights?"
245  return BiDijkStepRes::negative_weights;
246  }
247  } else if (auto search_w = seen.find(w);
248  search_w == std::end(seen) || vw_length < search_w->second) {
249  // Relax v-w edge
250  seen.insert_or_assign(w, vw_length);
251  fringe.push(std::make_pair(w, vw_length));
252 
253  // Build new path to w
254  predecessor[w] = v;
255 
256  // See if this path is better than the already discovered shortests path
257  if (direction == Direction::forward) {
258  compare_and_update_shortest_path(seen, other_seen, predecessor,
259  other_predecessor, s, t, w,
260  final_distance, final_path);
261  } else {
262  compare_and_update_shortest_path(other_seen, seen, other_predecessor,
263  predecessor, s, t, w, final_distance,
264  final_path);
265  }
266  }
267  }
268  // Everything went fine, go with other direction
269  return BiDijkStepRes::next;
270 }
