347  bool dominates(const Label &label) {
348 
349  auto node_n = label.get_node();
350  // If node_n is not in the skyline, then we have no labels to check
351  if (!contains(node_n)) {
352  return false;
353  }
354 
355  // For each of the paths p' we have similarity measure in 'label' we
356  // check if label.sim(p') is less than at least one of the labels in the
357  // skyline. If so, the skyline does NOT dominates 'label'.
358  for (const auto label_p : container[node_n]) {
359  LabelPtr tmpLabel = label_p;
360  bool skyline_dominates_label = true;
361 
362  for (int i = 0; i < static_cast<int>(label.get_num_paths()); ++i) {
363  if (label.get_similarity_with(i) < tmpLabel->get_similarity_with(i)) {
364  skyline_dominates_label = false;
365  break;
366  }
367  }
368 
369  if (skyline_dominates_label) {
370  return true;
371  }
372  }
373 
374  return false;
375  }

450  typename WeightMap>
451 bool update_label_similarity(Label &label, const Graph &G,
452  const EdgesMap &resEdges, const PathsMap &resPaths,
453  WeightMap &weight, double theta,
454  std::size_t step) {
455  using namespace boost;
456  bool below_sim_threshold = true;
457  auto tmpPath = label.get_path(G);
458  for (auto &e : tmpPath) {
459  auto search = resEdges.find(e);
460  // if tmpPath share an edge with any k-th shortest path, update the
461  // overlapping factor
462  if (search != std::end(resEdges)) {
463  for (auto index : search->second) {
464  if (static_cast<int>(index) > label.last_check() && index < step) {
465  label.get_similarity_with(index) += weight[e];
466 
467  // Check Lemma 1. The similarity between the candidate path and
468  // all the other k-shortest-paths must be less then theta
469  auto const &alt_path = resPaths[index];
470  auto alt_len = compute_length_from_edges(alt_path.begin(),
471  alt_path.end(), weight);
472  if (label.get_similarity_with(index) / alt_len > theta) {
473  below_sim_threshold = false;
474  break;
475  }
476  }
477  }
478  }
479  }
480  return below_sim_threshold;
