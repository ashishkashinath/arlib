403 template <typename Graph, typename WeightMap, typename DeletedEdgeMap,
404  typename Edge = edge_of_t<Graph>,
405  typename Length = length_of_t<Graph>>
406 int compute_priority(const Graph &G, const Edge &e, WeightMap &weight,
407  const DeletedEdgeMap &deleted_edge_map) {
408  using namespace boost;
409  using Vertex = typename graph_traits<Graph>::vertex_descriptor;
410  int priority = 0;
411  auto sources = std::vector<Vertex>{};
412  auto targets = std::vector<Vertex>{};
413 
414  auto a = source(e, G);
415  auto b = target(e, G);
416 
417  // Compute all the n_i nodes s.t. (n_i, a) is an incoming edge of a, with n_i
418  // != b
419  for (auto in_it = in_edges(a, G).first; in_it != in_edges(a, G).second;
420  ++in_it) {
421  auto n_i = source(*in_it, G);
422  if (n_i != b) {
423  sources.push_back(n_i);
424  }
425  }
426 
427  // Compute all the n_j nodes s.t. (b, n_j) is an outgoing edge of b, with n_j
428  // != a
429  for (auto out_it = out_edges(b, G).first; out_it != out_edges(b, G).second;
430  ++out_it) {
431  auto n_j = target(*out_it, G);
432  if (n_j != a) {
433  targets.push_back(n_j);
434  }
435  }
436 
437  // Get a graph with deleted edges filtered out
438  auto filter = edge_deleted_filter{deleted_edge_map};
439  const auto filtered_G = filtered_graph(G, filter);
440 
441  for (auto s_i : sources) {
442  for (auto t_i : targets) {
443  // Compute the shortest path from s_i to t_i
444  auto index = get(vertex_index, filtered_G);
445  auto predecessor_vec = std::vector<Vertex>(num_vertices(G), s_i);
446  auto predecessor =
447  make_iterator_property_map(predecessor_vec.begin(), index);
448  auto distance_vec = std::vector<Length>(num_vertices(G));
449  auto distance = make_iterator_property_map(distance_vec.begin(), index);
450 
451  auto rev_G = make_reverse_graph(filtered_G);
452  using RevEdge =
453  typename graph_traits<reverse_graph<Graph>>::edge_descriptor;
454  auto rev_weight = make_function_property_map<RevEdge>(
455  reverse_weight_functor{weight, filtered_G, rev_G});
456  auto rev_index = get(vertex_index, rev_G);
457 
458  try {
459  bidirectional_dijkstra(filtered_G, s_i, t_i, predecessor, distance,
460  weight, rev_G, rev_weight, rev_index);
461  if (shortest_path_contains_edge(s_i, t_i, e, filtered_G, predecessor)) {
462  ++priority;
463  }
464  } catch (target_not_found &ex) {
465  // In case t could not be found do nothing
466  std::cout << "Caught exception: " << ex.what() << "\n";
467  }
468  }
469  }
470 
471  return priority;
472 }

610 template <typename Graph, typename WeightMap, typename MultiPredecessorMap,
611  typename PriorityFunc, typename RoutingKernel, typename Terminator,
612  typename Vertex = vertex_of_t<Graph>>
613 void esx(const Graph &G, WeightMap const &weight,
614  MultiPredecessorMap &predecessors, Vertex s, Vertex t, int k,
615  double theta, PriorityFunc &&priority_fn,
616  RoutingKernel &routing_kernel, Terminator &&terminator) {
617  using namespace boost;
618  using Edge = typename graph_traits<Graph>::edge_descriptor;
619 
620  BOOST_CONCEPT_ASSERT((VertexAndEdgeListGraphConcept<Graph>));
621  BOOST_CONCEPT_ASSERT((LvaluePropertyMapConcept<WeightMap, Edge>));
622 
623  // P_LO set of k paths
624  auto resPathsEdges = std::vector<std::vector<Edge>>{};
625  auto resEdges = std::vector<std::unordered_set<Edge, boost::hash<Edge>>>{};
626 
627  BOOST_CONCEPT_ASSERT((VertexAndEdgeListGraphConcept<Graph>));
628  BOOST_CONCEPT_ASSERT((LvaluePropertyMapConcept<WeightMap, Edge>));
629 
630  // P_LO set of k paths
631  auto resPaths = std::vector<Path<Graph>>{};
632  // Compute shortest path from s to t
633  auto sp = details::compute_shortest_path(G, weight, s, t);
634  if (!sp) {
635  auto oss = std::ostringstream{};
636  oss << "Vertex " << t << " is unreachable from " << s;
637  throw details::target_not_found{oss.str()};
638  }
639 
640  // P_LO <-- {shortest path p_0(s, t)};
641  resPathsEdges.push_back(*sp);
642  resEdges.emplace_back(sp->begin(), sp->end());
643 
644  // If we need the shortest path only
645  if (k == 1) {
646  fill_multi_predecessor(resPathsEdges.begin(), resPathsEdges.end(), G,
647  predecessors);
648  return;
649  }
650 
651  // Every max-heap H_i is associated with p_i
652  using Priority = std::pair<Edge, int>;
653  using EdgePriorityQueue =
654  std::priority_queue<Priority, std::vector<Priority>,
655  details::EdgePriorityComparator<Edge>>;
656  auto edge_priorities = std::vector<EdgePriorityQueue>(k);
657 
658  // We keep a set of non-removable edges
659  auto dnr_edges = std::unordered_set<Edge, boost::hash<Edge>>{};
660 
661  // We keep a set of deleted-edges
662  auto deleted_edges = std::unordered_set<Edge, boost::hash<Edge>>{};
663 
664  // Compute lower bounds for AStar
665  // auto heuristic = details::distance_heuristic<Graph, Length>(G, t);
666 
667  // Initialize max-heap H_0 with the priority of each edge of the shortest path
668  priority_fn(*sp, edge_priorities, 0, G, weight, deleted_edges);
669 
670  auto overlaps = std::vector<double>(k, 0.0);
671  overlaps[0] = 1.0; // Set p_c overlap with sp (itself) to 1
672 
673  bool still_feasible = true;
674  while (resPathsEdges.size() < static_cast<std::size_t>(k) && still_feasible) {
675  std::ptrdiff_t p_max_idx = 0;
676  double overlap_ratio = 1.0;
677 
678  while (overlap_ratio >= theta) {
679  // Get the max overlapping path
680  auto max_it = std::max_element(std::begin(overlaps), std::end(overlaps));
681  p_max_idx = max_it - std::begin(overlaps);
682  overlap_ratio = *max_it;
683 
684  // Check if finding a result is feasible
685  still_feasible = details::check_feasibility(overlaps);
686  if (!still_feasible) {
687  break; // Stop the algorithm
688  }
689 
690  // Get e_tmp with higher priority from H_{p_max_idx}
691  auto e_tmp = edge_priorities[p_max_idx].top().first;
692 
693  // If edge is in DO-NOT-REMOVE edges set, continue
694  if (dnr_edges.find(e_tmp) != std::end(dnr_edges)) {
695  edge_priorities[p_max_idx].pop();
696  // Also, if H_{p_max_idx} queue is empty set its overlapping value to 0
697  if (edge_priorities[p_max_idx].empty()) {
698  overlaps[p_max_idx] = 0;
699  }
700  continue;
701  } else {
702  // Otherwise, add e_tmp to deleted edges
703  deleted_edges.insert(e_tmp);
704  }
705 
706  // The remainder code is the hot part of the algorithm. So we check here
707  // if the algorithm should terminate
708  if (terminator.should_stop()) {
709  throw terminator_stop_error{
710  "ESX terminated before completing due to a Terminator. Please "
711  "discard partial output."};
712  }
713 
714  // Compute p_tmp shortest path
715  auto p_tmp = routing_kernel(G, s, t, weight, deleted_edges);
716 
717  // If shortest path did not find a path
718  if (!p_tmp) {
719  move_to_dnr(e_tmp, deleted_edges, dnr_edges);
720  continue;
721  }
722 
723  // Remove e_tmp from H_{p_max_idx}
724  edge_priorities[p_max_idx].pop();
725 
726  // in cases there are no more edges to remove we set overlap to zero to
727  // avoid choosing from the same path again. A path the overlap of which is
728  // zero can never be chosen to remove its edges.
729  if (edge_priorities[p_max_idx].empty()) {
730  overlaps[p_max_idx] = 0;
731  } else {
732  const auto &alt_path = resEdges[p_max_idx];
733  overlaps[p_max_idx] = compute_similarity(*p_tmp, alt_path, weight);
734  }
735 
736  // Checking if the resulting path is valid
737  bool candidate_is_valid =
738  check_candidate_validity(*p_tmp, resEdges, weight, theta);
739  if (candidate_is_valid) {
740  // Add p_tmp to P_LO
741  resPathsEdges.emplace_back(*p_tmp);
742  resEdges.emplace_back(p_tmp->begin(), p_tmp->end());
743 
744  // Set p_c overlap with itself to 1
745  std::ptrdiff_t p_c_idx = resPathsEdges.size() - 1;
746  overlaps[p_c_idx] = 1.0;
747 
748  // Initialize max-heap H_i with the priority of each edge of new
749  // alternative path
750  priority_fn(*p_tmp, edge_priorities, p_c_idx, G, weight, deleted_edges);
751  break; // From inner while loop
752  }
753  }
754  }
755 
756  // Beforer returning, populate predecessors map
757  fill_multi_predecessor(resPathsEdges.begin(), resPathsEdges.end(), G,
758  predecessors);
759 }
