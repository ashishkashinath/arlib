457 template <typename Graph, typename DistanceMap, typename PMap,
458  typename Vertex = vertex_of_t<Graph>,
459  typename Edge = edge_of_t<Graph>,
460  typename Length = length_of_t<Graph>>
461 void penalize_candidate_path(const std::vector<Edge> &candidate, const Graph &G,
462  Vertex s, Vertex t, double p, double r,
463  penalty_functor<PMap> &penalty,
464  const DistanceMap &distance_s,
465  const DistanceMap &distance_t,
466  PenBoundsMap<Edge> &penalty_bounds,
467  int bound_limit) {
468  using namespace boost;
469 
470  // Keep track of candidate vertices to exclude them from incoming/outgoing
471  // edges update
472  auto candidate_vertices = std::unordered_set<Vertex, boost::hash<Vertex>>{};
473  auto candidate_edges = std::unordered_set<Edge, boost::hash<Edge>>{};
474 
475  for (const auto &e : candidate) {
476  candidate_edges.insert(e);
477 
478  auto u = source(e, G);
479  auto v = target(e, G);
480  candidate_vertices.insert(u);
481  candidate_vertices.insert(v);
482  }
483 
484  for (auto &e : candidate_edges) {
485  auto u = source(e, G);
486  auto v = target(e, G);
487 
488  // Check if 'e' is already part of the alternative graph
489  if (auto search = penalty_bounds.find(e);
490  search != std::end(penalty_bounds)) {
491  // If so, penalize only if limit isnt reached
492  auto n_updates = search->second;
493  if (n_updates < bound_limit) {
494  penalty[e] += p * penalty[e];
495  ++penalty_bounds[e];
496  }
497  } else { // Penalize and create a nb_updates counter for e
498  penalty[e] += p * penalty[e];
499  penalty_bounds.insert({e, 1});
500  }
501 
502  // Update incoming edges
503  for (auto [it, end] = in_edges(u, G); it != end; ++it) {
504  auto a = source(*it, G);
505 
506  // Incoming edge (a, u) is updated only if 'a' is not part of candidate
507  // path
508  if (candidate_vertices.find(a) == std::end(candidate_vertices)) {
509  // Check if '*it' is already part of the alternative graph
510  if (auto search = penalty_bounds.find(*it);
511  search != std::end(penalty_bounds)) {
512  auto n_updates = search->second;
513  // penalize only if limit isnt reached
514  if (n_updates < bound_limit) {
515  auto closeness = distance_t[u] / distance_t[s];
516  auto pen_factor = 0.1 + r * closeness;
517  penalty[*it] += pen_factor * penalty[*it];
518  ++penalty_bounds[*it];
519  }
520  } else {
521  // Else, just update it
522  auto closeness = distance_t[u] / distance_t[s];
523  auto pen_factor = 0.1 + r * closeness;
524  penalty[*it] += pen_factor * penalty[*it];
525  }
526  }
527  }
528 
529  // Update outgoing edges
530  for (auto [it, end] = out_edges(v, G); it != end; ++it) {
531  auto b = target(*it, G);
532 
533  // Outgoing edge (v, b) is updated only if 'b' is not part of candidate
534  // path
535  if (candidate_vertices.find(b) == std::end(candidate_vertices)) {
536  // Check if '*it' is already part of the alternative graph
537  if (auto search = penalty_bounds.find(*it);
538  search != std::end(penalty_bounds)) {
539  auto n_updates = search->second;
540  // penalize only if limit isnt reached
541  if (n_updates < bound_limit) {
542  auto closeness = distance_s[v] / distance_s[t];
543  auto pen_factor = 0.1 + r * closeness;
544  penalty[*it] += pen_factor * penalty[*it];
545  ++penalty_bounds[*it];
546  }
547  } else {
548  // Else, just update it
549  auto closeness = distance_s[v] / distance_s[t];
550  auto pen_factor = 0.1 + r * closeness;
551  penalty[*it] += pen_factor * penalty[*it];
552  }
553  }
554  }
555  }
556 }

558 template <typename Graph, typename WeightMap, typename MultiPredecessorMap,
559  typename RoutingKernel, typename Terminator,
560  typename Vertex = vertex_of_t<Graph>>
561 void penalty(const Graph &G, WeightMap const &original_weight,
562  MultiPredecessorMap &predecessors, Vertex s, Vertex t, int k,
563  double theta, double p, double r, int max_nb_updates,
564  int max_nb_steps, RoutingKernel &routing_kernel,
565  Terminator &&terminator) {
566  using namespace boost;
567  using Edge = typename graph_traits<Graph>::edge_descriptor;
568  using Length = typename boost::property_traits<typename boost::property_map<
569  Graph, boost::edge_weight_t>::type>::value_type;
570 
571  BOOST_CONCEPT_ASSERT((VertexAndEdgeListGraphConcept<Graph>));
572  BOOST_CONCEPT_ASSERT((LvaluePropertyMapConcept<WeightMap, Edge>));
573 
574  // P_LO set of k paths
575  auto resPathsEdges = std::vector<std::vector<Edge>>{};
576  auto resEdges = std::vector<std::unordered_set<Edge, boost::hash<Edge>>>{};
577 
578  // P_LO set of k paths
579  auto resPaths = std::vector<Path<Graph>>{};
580  // Make a local weight map to avoid modifying existing graph.
581  auto pen_fctor = details::penalty_functor{original_weight};
582 
583  // Compute shortest path from s to t
584  auto distance_s = std::vector<Length>(num_vertices(G));
585  auto distance_t = std::vector<Length>(num_vertices(G));
586  auto sp = dijkstra_shortest_path_two_ways(G, s, t, distance_s, distance_t);
587  if (!sp) {
588  auto oss = std::ostringstream{};
589  oss << "Vertex " << t << " is unreachable from " << s;
590  throw details::target_not_found{oss.str()};
591  }
592 
593  // P_LO <-- {shortest path p_0(s, t)};
594  resPathsEdges.push_back(*sp);
595  resEdges.emplace_back(sp->begin(), sp->end());
596 
597  // If we need the shortest path only
598  if (k == 1) {
599  fill_multi_predecessor(resPathsEdges.begin(), resPathsEdges.end(), G,
600  predecessors);
601  return;
602  }
603 
604  // Initialize map for penalty bounds
605  auto penalty_bounds = std::unordered_map<Edge, int, boost::hash<Edge>>{};
606 
607  // Penalize sp edges
608  penalize_candidate_path(*sp, G, s, t, p, r, pen_fctor, distance_s, distance_t,
609  penalty_bounds, max_nb_updates);
610 
611  int step = 0;
612  using Index = std::size_t;
613  while (resPathsEdges.size() < static_cast<Index>(k) && step < max_nb_steps) {
614  // The remainder code is the hot part of the algorithm. So we check here
615  // if the algorithm should terminate
616  if (terminator.should_stop()) {
617  throw terminator_stop_error{
618  "Penalty terminated before completing due to a Terminator. Please "
619  "discard partial output."};
620  }
621 
622  auto p_tmp = routing_kernel(G, s, t, pen_fctor);
623 
624  // Penalize p_tmp edges
625  penalize_candidate_path(*p_tmp, G, s, t, p, r, pen_fctor, distance_s,
626  distance_t, penalty_bounds, max_nb_updates);
627  ++step;
628 
629  // If p_tmp is sufficiently dissimilar to other alternative paths, accept it
630  bool is_valid_path = true;
631  for (const auto &alt_path : resEdges) {
632  if (compute_similarity(*p_tmp, alt_path, original_weight) > theta) {
633  is_valid_path = false;
634  break;
635  }
636  }
637 
638  if (is_valid_path) {
639  resPathsEdges.push_back(*p_tmp);
640  resEdges.emplace_back(p_tmp->begin(), p_tmp->end());
641  }
642  }
643 
644  // Beforer returning, populate predecessors map
645  fill_multi_predecessor(resPathsEdges.begin(), resPathsEdges.end(), G,
646  predecessors);
647 }
