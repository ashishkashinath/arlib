99  MultiPredecessorMap &predecessors, Vertex s, Vertex t, int k,
100  double theta, Terminator &&terminator = Terminator{}) {
101  using namespace boost;
102  using Edge = typename graph_traits<Graph>::edge_descriptor;
103  using Length = typename boost::property_traits<WeightMap>::value_type;
104 
105  BOOST_CONCEPT_ASSERT((VertexAndEdgeListGraphConcept<Graph>));
106  BOOST_CONCEPT_ASSERT((LvaluePropertyMapConcept<WeightMap, Edge>));
107  BOOST_CONCEPT_ASSERT(
108  (ReadablePropertyMapConcept<MultiPredecessorMap, Vertex>));
109 
110  auto resPathsEdges = std::vector<std::vector<Edge>>{};
111 
112  // resEdges keeps track of the edges that make the paths in resPaths and which
113  // path includes it.
114  using resPathIndex = typename decltype(resPathsEdges)::size_type;
115  auto resEdges =
116  std::unordered_map<Edge, std::vector<resPathIndex>, boost::hash<Edge>>{};
117 
118  // Min-priority queue
119  using Label = details::OnePassLabel<Graph, Length>;
120  using LabelPtr = std::unique_ptr<Label>;
121  auto Q =
122  std::priority_queue<Label *, std::vector<Label *>,
123  details::OnePassPlusASComparator<Graph, Length>>{};
124  auto created_labels = std::vector<LabelPtr>{};
125 
126  // Skyline for dominance checkind (Lemma 2)
127  auto skyline = details::SkylineContainer<Graph, Length>{};
128 
129  // Compute lower bounds for AStar
130  auto lower_bounds = details::distance_from_target<Length>(G, t);
131 
132  // Compute shortest path from s to t
133  auto sp_path = details::compute_shortest_path(G, weight, s, t);
134  if (!sp_path) {
135  auto oss = std::ostringstream{};
136  oss << "Vertex " << t << " is unreachable from " << s;
137  throw details::target_not_found{oss.str()};
138  }
139 
140  // P_LO <-- {shortest path p_0(s, t)};
141  resPathsEdges.push_back(*sp_path);
142  resPathIndex paths_count = 1;
143 
144  // If we need the shortest path only
145  if (k == 1) {
146  details::fill_multi_predecessor(resPathsEdges.begin(), resPathsEdges.end(),
147  G, predecessors);
148  return;
149  }
150 
151  // For each edge in the candidate path, we check if it's already in any of the
152  // resPaths. If not, we add it to resEdges. If yes, we keep track of which
153  // path includes it.
154  details::update_res_edges(*sp_path, resEdges, paths_count);
155 
156  // Initialize min-priority queue Q with <s, empty_set>
157  auto init_label =
158  std::make_unique<Label>(s, 0, lower_bounds[s], k, paths_count - 1);
159  Q.push(init_label.get());
160  created_labels.push_back(std::move(init_label));
161 
162  // While Q is not empty
163  while (!Q.empty()) {
164  // The remainder code is the hot part of the algorithm. So we check here
165  // if the algorithm should terminate
166  if (terminator.should_stop()) {
167  throw terminator_stop_error{
168  "OnePass+ terminated before completing due to a Terminator. Please "
169  "discard partial output."};
170  }
171 
172  // Current path
173  auto label = Q.top();
174  Q.pop();
175 
176  // Perform lazy update of the similairty vector of 'label', since new paths
177  // might have been added to P_LO from the time this 'label' was pushed into
178  // priority queue.
179  if (label->is_outdated(paths_count - 1)) {
180  bool below_sim_threshold = details::update_label_similarity(
181  *label, G, resEdges, resPathsEdges, weight, theta, paths_count);
182 
183  label->set_last_check(paths_count - 1); // Update last check time step
184  if (!below_sim_threshold) {
185  continue; // Skip candidate path
186  }
187  }
188 
189  // If we found the target node
190  if (label->get_node() == t) {
191  // Build the new k-th shortest path
192  resPathsEdges.push_back(label->get_path(G));
193 
194  auto &tmpPath = resPathsEdges.back();
195  ++paths_count;
196 
197  if (static_cast<int>(paths_count) == k) { // we found k paths. End.
198  // Add computed alternatives to resPaths
199  details::fill_multi_predecessor(resPathsEdges.begin(),
200  resPathsEdges.end(), G, predecessors);
201  break;
202  }
203 
204  // For each edge in the candidate path see if it's already in any of the
205  // P_LO paths. If not, add it to the resEdges. If so, keep track of which
206  // path includes it
207  details::update_res_edges(tmpPath, resEdges, paths_count);
208 
209  } else { // Expand Search
210  if (skyline.dominates(*label)) {
211  continue; // Prune path by Lemma 2
212  }
213 
214  skyline.insert(label);
215  auto node_n = label->get_node();
216  // For each outgoing edge
217  for (auto adj_it = adjacent_vertices(node_n, G).first;
218  adj_it != adjacent_vertices(node_n, G).second; ++adj_it) {
219  // Expand path
220  auto c_edge = edge(node_n, *adj_it, G).first;
221  auto c_label =
222  details::expand_path(label, *adj_it, lower_bounds[*adj_it],
223  weight[c_edge], paths_count - 1);
224 
225  // Check for acyclicity
226  bool acyclic = label->is_path_acyclic(*adj_it);
227 
228  if (acyclic) {
229  auto c_similarity_map = label->get_similarity_map();
230 
231  // Check Lemma 1 for similarity thresholding
232  bool below_sim_threshold = details::is_below_sim_threshold(
233  c_edge, c_similarity_map, theta, resEdges, resPathsEdges, weight);
234 
235  if (below_sim_threshold) {
236  c_label->set_similarities(std::begin(c_similarity_map),
237  std::end(c_similarity_map));
238  Q.push(c_label.get());
239  created_labels.push_back(std::move(c_label));
240  }
241  }
242  }
243  }
244  }
245 
246  if (static_cast<int>(paths_count) != k) {
247  // Add computed alternatives to resPaths
248  details::fill_multi_predecessor(resPathsEdges.begin(), resPathsEdges.end(),
249  G, predecessors);
250  }
251 }
